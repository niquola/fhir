Example of medicationadministration (id = "example")

Raw XML

<MedicationAdministration xmlns="http://hl7.org/fhir"> 
<text>
    <status value="generated"/>
    <div xmlns="http://www.w3.org/1999/xhtml">
      <p>Penicillin VK 10ml suspension administered by oral route at 14:30 on 1 June 2012</p>
      <p>on the authority of prescription ref: example</p>
      <p>to patient ref: a23</p>
    </div>
  </text> 
<!--      -->
  <status value="completed"/>
  <patient>
    <type value="Patient"/>
    <reference value="patient/@example"/>
  </patient>
  <practitioner>
    <type value="Practitioner"/>
    <reference value="practitioner/@example"/>
  </practitioner>
  <prescription>
    <type value="MedicationPrescription"/>
    <reference value="medicationprescription/@example"/>
  </prescription>
  <whenGiven>
  <start value="2012-06-01T14:30:00"/>
  <end value="2012-06-01T14:30:00"/>
</whenGiven>
  <medication>
    <type value="Medication"/>
    <reference value="medication/@example"/> 
  </medication>
  <dosage>
    <route>
      <coding>
        <system value="http://snomed.info/id"/>
        <code value="363743006"/>
        <display value="oral administration of treatment"/>
      </coding>  
    </route>
    <quantity>
      <value value="10"/>
      <units value="ml"/>
      <system value="http://unitsofmeasure.org"/>
      <code value="ml"/>
    </quantity>
  </dosage>
</MedicationAdministration>
